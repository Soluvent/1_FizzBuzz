```java
private static boolean canBalance(int[] arr) {
    int totalSum = 0;
    for (int num : arr) {
        totalSum += num;
    }
    int leftSum = 0;
    for (int j : arr) {
        leftSum += j;
        int rightSum = totalSum - leftSum;
        if (leftSum == rightSum) {
            return true;
        }
    }
    return false;
}
```
